import unittest
from EmotionDetection.emotion_detection import emotion_detector

class TestEmotionDetection(unittest.TestCase):
    def test_emotion_detector(self):
        # Test Case 1: Dominant Emotion is Joy
        self.assertEqual(
            emotion_detector("I am glad this happened")['dominant_emotion'], 
            'joy'
        )
        
        # Test Case 2: Dominant Emotion is Anger
        self.assertEqual(
            emotion_detector("I am really mad at you")['dominant_emotion'], 
            'anger'
        )
        
        # Test Case 3: Dominant Emotion is Sadness
        self.assertEqual(
            emotion_detector("I am so sad about this")['dominant_emotion'], 
            'sadness'
        )
        
        # Test Case 4: Dominant Emotion is Fear
        self.assertEqual(
            emotion_detector("I am really afraid that this will happen")['dominant_emotion'], 
            'fear'
        )
        
        # Test Case 5: Dominant Emotion is Disgust
        self.assertEqual(
            emotion_detector("I feel disgusted just hearing about this")['dominant_emotion'], 
            'disgust'
        )

if __name__ == '__main__':
    unittest.main()
